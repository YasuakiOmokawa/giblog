# Development

# Create new website
rm -rf mysite && make && perl -Mblib script/giblog new mysite
rm -rf mysite_zemi && make && perl -Mblib script/giblog new_zemi mysite_zemi
rm -rf mysite_test && make && perl -Mblib script/giblog new_test mysite_test

# Serve
